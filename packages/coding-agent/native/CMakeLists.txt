cmake_minimum_required(VERSION 3.14)

project(bone_embed LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(BONE_CRISPEMBED_INCLUDE_DIR "" CACHE PATH "CrispEmbed installed include directory")
set(BONE_CRISPEMBED_LIBRARY "" CACHE FILEPATH "CrispEmbed shared or import library")
set(BONE_NODE_API_INCLUDE_DIR "" CACHE PATH "Node-API include directory containing node_api.h")
set(BONE_NODE_LIBRARY "" CACHE FILEPATH "Node import library; required on Windows")

if(NOT EXISTS "${BONE_CRISPEMBED_INCLUDE_DIR}/crispembed.h")
	message(FATAL_ERROR "BONE_CRISPEMBED_INCLUDE_DIR must contain crispembed.h")
endif()
if(NOT EXISTS "${BONE_CRISPEMBED_LIBRARY}")
	message(FATAL_ERROR "BONE_CRISPEMBED_LIBRARY must point to the CrispEmbed library")
endif()
if(NOT EXISTS "${BONE_NODE_API_INCLUDE_DIR}/node_api.h")
	message(FATAL_ERROR "BONE_NODE_API_INCLUDE_DIR must contain node_api.h")
endif()
if(WIN32 AND NOT EXISTS "${BONE_NODE_LIBRARY}")
	message(FATAL_ERROR "BONE_NODE_LIBRARY must point to node.lib on Windows")
endif()

add_library(bone-embed MODULE bone-embed-addon.cpp)
target_include_directories(bone-embed PRIVATE "${BONE_CRISPEMBED_INCLUDE_DIR}" "${BONE_NODE_API_INCLUDE_DIR}")
target_link_libraries(bone-embed PRIVATE "${BONE_CRISPEMBED_LIBRARY}")
set_target_properties(bone-embed PROPERTIES PREFIX "" SUFFIX ".node")

if(APPLE)
	set_target_properties(bone-embed PROPERTIES BUILD_RPATH "@loader_path" INSTALL_RPATH "@loader_path")
	target_link_options(bone-embed PRIVATE "-undefined" "dynamic_lookup")
elseif(UNIX)
	set_target_properties(bone-embed PROPERTIES BUILD_RPATH "$ORIGIN" INSTALL_RPATH "$ORIGIN")
elseif(WIN32)
	target_link_libraries(bone-embed PRIVATE "${BONE_NODE_LIBRARY}")
endif()

install(TARGETS bone-embed LIBRARY DESTINATION . RUNTIME DESTINATION .)
